    <script>

        const { createApp, ref, computed } = Vue;


        createApp({

            setup() {

                // 初始化候選人資料 (比照圖片)

                const candidates = ref([

                    { id: 1, name: "梁秀娟", gender: "女", age: 51, church: "大溪", education: "大學", title: "教牧", isEssential: true, checked: true, votes: 12 },

                    { id: 2, name: "王鴻志", gender: "男", age: 52, church: "大溪", education: "大專", title: "宣道", isEssential: false, checked: false, votes: 8 },

                    { id: 3, name: "張蓉惠", gender: "女", age: 26, church: "大溪", education: "大學", title: "教育", isEssential: false, checked: false, votes: 5 },

                    { id: 4, name: "楊忠生", gender: "男", age: 52, church: "大溪", education: "專科", title: "事務", isEssential: false, checked: false, votes: 9 },

                    { id: 5, name: "林茜葦", gender: "女", age: 48, church: "大溪", education: "大專", title: "資訊", isEssential: false, checked: false, votes: 4 },

                    { id: 6, name: "鍾霖恩", gender: "男", age: 42, church: "大溪", education: "大學", title: "會計", isEssential: false, checked: false, votes: 7 },

                    { id: 7, name: "黃琬珊", gender: "女", age: 40, church: "大溪", education: "大學", title: "出納", isEssential: false, checked: false, votes: 3 }

                ]);


                // 計算目前勾選了幾個人

                const selectedCount = computed(() => {

                    return candidates.value.filter(c => c.checked).length;

                });


                // 計算總票數 (以教牧的票數作為總票數基準，因為每張票必有教牧)

                const totalVotes = computed(() => {

                    const essential = candidates.value.find(c => c.isEssential);

                    return essential ? essential.votes : 0;

                });


                // 排序後的候選人（按票數從高到低排，開票畫面更直觀）

                const sortedCandidates = computed(() => {

                    return [...candidates.value].sort((a, b) => b.votes - a.votes);

                });


                // 計算得票百分比

                const getPercentage = (votes) => {

                    if (totalVotes.value === 0) return 0;

                    return ((votes / totalVotes.value) * 100).toFixed(1);

                };


                // 投票送出事件

                const submitVote = () => {

                    if (selectedCount.value !== 3) return;


                    // 模擬寫入資料庫：將勾選的人票數 +1

                    candidates.value.forEach(c => {

                        if (c.checked) {

                            c.votes += 1;

                        }

                    });


                    alert("🎉 投票成功！感謝您的參與。");


                    // 重設選票狀態（教牧保持預設勾選，其餘清空）

                    candidates.value.forEach(c => {

                        if (!c.isEssential) c.checked = false;

                    });

                };


                return {

                    candidates,

                    selectedCount,

                    totalVotes,

                    sortedCandidates,

                    getPercentage,

                    submitVote

                };

            }

        }).mount('#app');

    </script>
